  - ragged2e;   
  - parskip;   
  - hyperref;   
  - textpos;   
  - titlesec;   
  - tikz;   
  - xcolor;   
  - multirow;   
  - xargs;   
  - tcolorbox;   
  - enumitem;   
  - ifthen.
